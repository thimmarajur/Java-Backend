-----DataSource Config inside core of service --------
Class is annotated with @Configuration

{
    @Bean(name = "masmysql")
    @ConfigurationProperties("masmysql.datasource")
    public DataSource masmysqlDataSource() {
        return DataSourceBuilder.create().build();
    }

    @Bean(name = "masmysqlTemplate")
    public JdbcTemplate masmysqlTemplate(@Qualifier("masmysql") DataSource ds) {
        return new JdbcTemplate(ds);
    }
	
}	
	

-----------DBUtil---------------
	@Autowired
	@Qualifier("masmysqlTemplate")
    JdbcTemplate masmysqlTemplate;

	public JdbcTemplate getTemplate() {
		
		JdbcTemplate jdbcTemplate = masmysqlTemplate;
		return jdbcTemplate;
		
	}
